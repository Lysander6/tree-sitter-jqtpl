================================
Output directives nested in if-s
================================

<div>
  {{if hello}}
    <p>{{= hello}}</p>
  {{else hi}}
    <p>{{= hi}}</p>
  {{else}}
    <p>{{= goodbye}}</p>
  {{/if}}
</div>

-----

(template
  (content)
  (if_directive
    condition: (code)
    consequence: (content)
    consequence: (output_directive
      (code))
    consequence: (content)
    alternative: (else_directive
      condition: (code)
      consequence: (content)
      consequence: (output_directive
        (code))
      consequence: (content)
      alternative: (else_directive
      consequence: (content)
      consequence: (output_directive
        (code))
      consequence: (content))))
  (content))
